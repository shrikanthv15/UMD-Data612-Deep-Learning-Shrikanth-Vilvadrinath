## P2-09: scoring.py -- pixel_anomaly_map

from pytorch_msssim import ssim

def pixel_anomaly_map(
    originals: Tensor,       # (B, 3, H, W) in [-1, 1]
    reconstructions: Tensor, # (B, 3, H, W) in [-1, 1]
    method: str = "ssim",    # "ssim" | "l2" | "lpips"
) -> Tensor:
    """
    Compute per-pixel anomaly score.
    
    For SSIM: anomaly = 1 - SSIM_map (higher = more anomalous)
    For L2: anomaly = ||orig - recon||^2 per pixel, averaged over channels
    For LPIPS: use lpips library (requires model download)
    
    Returns: (B, 1, H, W) anomaly map in [0, 1]
    """
